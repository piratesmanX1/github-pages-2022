git status // check the status of the local workfile compare to the repository's
git add . // add as in adding all the files into commit ; '.'' stands for all
git commit -m initialcommit // commit your local working files changes into Github, description will be named as 'initial commit' via 'git commit -m'
git push origin main // push the committed files into the selected branch, in this case it will be 'main' branch
git checkout -b styling // creating a new branch name 'styling' via 'git checkout -b'
git checkout -b site
git push origin site // we did not need to commit as we did not changed any file within the repo, so we can just straightaway push it when creating the branch
